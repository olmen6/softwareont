29static uint8_t color_from_string(const char *color)
30{
31 while (*color == ' ')
32 color++;
33 // Kleuren naam mapping
34 if (strcmp(color, "wit") == 0) return VGA_COL_WHITE;
35 if (strcmp(color, "zwart") == 0) return VGA_COL_BLACK;
36 if (strcmp(color, "rood") == 0) return VGA_COL_RED;
37 if (strcmp(color, "groen") == 0) return VGA_COL_GREEN;
38 if (strcmp(color, "blauw") == 0) return VGA_COL_BLUE;
39 if (strcmp(color, "geel") == 0) return VGA_COL_YELLOW;
40 if (strcmp(color, "cyaan") == 0) return VGA_COL_CYAN;
41 if (strcmp(color, "magenta") == 0) return VGA_COL_MAGENTA;
42 // fallback: gebruik geel
43
44 return VGA_COL_YELLOW;
45}
