205void TIM2_IRQHandler(void)
206{
207 /* USER CODE BEGIN TIM2_IRQn 0 */
208
209 /* USER CODE END TIM2_IRQn 0 */
210 HAL_TIM_IRQHandler(&htim2);
211 /* USER CODE BEGIN TIM2_IRQn 1 */
212 __HAL_TIM_CLEAR_IT(&htim2, TIM_IT_CC3); // uncomment HAL_TIM_IRQHandler(&htim2); ^^
213
214 VGA.hsync_cnt++;
215 if (VGA.hsync_cnt >= VGA_VSYNC_PERIODE)
216 {
217 // -----------
218 VGA.hsync_cnt = 0;
219 // Adresspointer first dot
220 VGA.start_adr = (uint32_t)(&VGA_RAM1[0]);
221 }
222
223 // HSync-Pixel
224 GPIOB->BSRR = (VGA.hsync_cnt < VGA_VSYNC_IMP) ? VGA_VSYNC_Pin << 16u: VGA_VSYNC_Pin;
225
226 // Test for DMA start
227 if((VGA.hsync_cnt >= VGA_VSYNC_BILD_START) && (VGA.hsync_cnt <= VGA_VSYNC_BILD_STOP))
228 {
229 // after FP start => DMA Transfer
230
231 // DMA2 init
232 DMA2_Stream5->CR = VGA.dma2_cr_reg;
233 // set adress
234 DMA2_Stream5->M0AR = VGA.start_adr;
235 // Timer1 start
236 TIM1->CR1 |= TIM_CR1_CEN; // __HAL_TIM_ENABLE(&htim1); // too slow?
237 // DMA2 enable
238 __HAL_DMA_ENABLE(&hdma_tim1_up);
239 // Test Adrespointer for high
240 if(VGA.hsync_cnt & 0x01)
241 VGA.start_adr += (VGA_DISPLAY_X + 1); // inc after Hsync
242 }
243 /* USER CODE END TIM2_IRQn 1 */
244}

249void USART2_IRQHandler(void)
250{
251 /* USER CODE BEGIN USART2_IRQn 0 */
252
253 // Store the byte we received on the UART
254 char uart_char = USART2->DR;
255
256 //Ignore the '\n' character
257 if(uart_char != LINE_FEED)
258 {
259 //Check for CR or a dot
260 // There was a small bug in the terminal program.
261 // By terminating your message with a dot you can ignore the CR (Enter) character
262 if((uart_char == CARRIAGE_RETURN) || (uart_char == '.'))
263 {
264 // Null-terminate the line so parsers don't read leftover characters
265 input.line_rx_buffer[input.char_counter] = '\0';
266 input.command_execute_flag = TRUE;
267 // Store the message length for processing
268 input.msglen = input.char_counter;
269 // Reset the counter for the next line
270 input.char_counter = 0;
271 //Gently exit interrupt
272 }
273 else
274 {
275 input.command_execute_flag = FALSE;
276 input.line_rx_buffer[input.char_counter] = uart_char;
277 input.char_counter++;
278 }
279 }
280
281 /* USER CODE END USART2_IRQn 0 */
282 HAL_UART_IRQHandler(&huart2);
283 /* USER CODE BEGIN USART2_IRQn 1 */
284
285 /* USER CODE END USART2_IRQn 1 */
286}
