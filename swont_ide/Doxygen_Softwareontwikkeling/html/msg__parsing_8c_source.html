33uint8_t parse_msg(char *msg, argList *args)
34{
35 if (msg == NULL || args == NULL)
36 {
37 Error_t err = { .layer = LAYER_APP, .code = ERR_PARAM, .module = "MsgParser", .msg = "parse_msg: NULL pointer" };
38 Error_Report(&err);
39 printing_done_flag = TRUE; // Must reset flag on error
40 return 0;
41 }
42
43 args->count = 0;
44 // Don't touch printing_done_flag here! Only API layer manages it
45
46 // Make a working copy since strtok modifies the original
47 char msg_copy[1024];
48 strncpy(msg_copy, msg, sizeof(msg_copy) - 1);
49 msg_copy[sizeof(msg_copy) - 1] = '\0';
50
51 char *tok = strtok(msg_copy, TOKEN);
52
53 if (tok == NULL){
54 Error_t err = { .layer = LAYER_APP, .code = ERR_PARAM, .module = "MsgParser", .msg = "geen tokens in bericht" };
55 Error_Report(&err);
56 printing_done_flag = TRUE; // Must reset flag on error
57 return (0);
58 }
59
60 while (tok && args->count < MAX_ARG_COUNT){
61 // Copy token string into the struct (safe copy)
62 strncpy(args->tokens[args->count], tok, MAX_TOKEN_LEN - 1);
63 args->tokens[args->count][MAX_TOKEN_LEN - 1] = '\0'; // Ensure null termination
64 args->count++;
65 tok = strtok(NULL, TOKEN);
66 }
67 return (1);
68}

74uint8_t process_msg(const argList *args)
75{
76 if (args == NULL || args->count == 0)
77 {
78 Error_t err = { .layer = LAYER_APP, .code = ERR_PARAM, .module = "MsgParser", .msg = "process_msg: geen argumenten" };
79 Error_Report(&err);
80 printing_done_flag = TRUE;
81 return (0);
82 }
83
84
85 const char *cmd = args->tokens[0];
86
87 if (cmd == NULL)
88 {
89 Error_t err = { .layer = LAYER_APP, .code = ERR_PARAM, .module = "MsgParser", .msg = "process_msg: commando NULL" };
90 Error_Report(&err);
91 printing_done_flag = TRUE; // Ensure flag is reset on error
92 return 0;
93 }
94
95 int i;
96 for (i = 0; commands[i].name != NULL; i++) {
97 if (strcmp(cmd, commands[i].name) == 0) {
98 // Command found, execute it. API layer will handle printing_done_flag
99 return commands[i].fn((argList *)args);
100 }
101 }
102
103 // Unknown command
104 Error_t err = { .layer = LAYER_APP, .code = ERR_PARAM, .module = "MsgParser", .msg = "onbekend commando" };
105 Error_Report(&err);
106 printing_done_flag = TRUE; // Reset flag on unrecognized command
107 return (0);
108}
